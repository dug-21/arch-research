// Use IIS Manager Machine Key feature or generate programmatically
using System.Web.Configuration;
using System.Security.Cryptography;

public static class SecureMachineKeyGenerator
{
    public static string GenerateSecureKeys()
    {
        // Generate 128-byte validation key (1024 bits)
        byte[] validationKeyBytes = new byte[128];
        using (var rng = RandomNumberGenerator.Create())
        {
            rng.GetBytes(validationKeyBytes);
        }
        string validationKey = BytesToHex(validationKeyBytes);
        
        // Generate 48-byte decryption key (384 bits for 3DES)
        byte[] decryptionKeyBytes = new byte[48];
        using (var rng = RandomNumberGenerator.Create())
        {
            rng.GetBytes(decryptionKeyBytes);
        }
        string decryptionKey = BytesToHex(decryptionKeyBytes);
        
        return $@"
<machineKey 
  validationKey=""{validationKey}"" 
  decryptionKey=""{decryptionKey}"" 
  validation=""HMACSHA256"" 
  decryption=""3DES"" />";
    }
    
    private static string BytesToHex(byte[] bytes)
    {
        return BitConverter.ToString(bytes).Replace("-", "");
    }
}

// Global.asax.cs security enhancements
public class Global : HttpApplication
{
    protected void Application_Start()
    {
        // Remove server headers
        MvcHandler.DisableMvcResponseHeader = true;
    }
    
    protected void Application_PreSendRequestHeaders()
    {
        Response.Headers.Remove("Server");
        Response.Headers.Remove("X-Powered-By");
        Response.Headers.Remove("X-AspNet-Version");
    }
    
    protected void Application_BeginRequest()
    {
        // Force HTTPS
        if (!Request.IsSecureConnection && !Request.IsLocal)
        {
            string redirectURL = Request.Url.ToString().Replace("http:", "https:");
            Response.Redirect(redirectURL, false);
            HttpContext.Current.ApplicationInstance.CompleteRequest();
        }
        
        // Additional security validation
        ValidateRequestSecurity();
    }
    
    private void ValidateRequestSecurity()
    {
        // Check for suspicious patterns
        if (Request.Form["__VIEWSTATE"] != null)
        {
            string viewState = Request.Form["__VIEWSTATE"];
            
            // Log large ViewState (potential attack)
            if (viewState.Length > 100000) // 100KB threshold
            {
                LogSecurityEvent($"Large ViewState detected: {viewState.Length} bytes from {Request.UserHostAddress}");
            }
        }
        
        // Validate User-Agent for known attack patterns
        if (string.IsNullOrEmpty(Request.UserAgent) || 
            Request.UserAgent.Contains("sqlmap") ||
            Request.UserAgent.Contains("nmap"))
        {
            LogSecurityEvent($"Suspicious User-Agent: {Request.UserAgent} from {Request.UserHostAddress}");
        }
    }
}

public class ViewStateSecurityAnalyzer
{
    public ViewStateAnalysisResult AnalyzeViewState(string viewStateData)
    {
        var result = new ViewStateAnalysisResult();
        
        try
        {
            // Decode Base64
            byte[] viewStateBytes = Convert.FromBase64String(viewStateData);
            result.ViewStateSizeBytes = viewStateBytes.Length;
            result.ViewStateSizeKB = Math.Round(viewStateBytes.Length / 1024.0, 2);
            
            // Analyze structure (simplified)
            result.ContainsControlState = viewStateData.Contains("ctl");
            result.ContainsEventValidation = viewStateData.Contains("__EVENTVALIDATION");
            
            // Security assessment
            if (result.ViewStateSizeKB > 50)
            {
                result.SecurityRisk = "HIGH - Large ViewState (>50KB)";
                result.Recommendations.Add("Implement ViewState minimization");
                result.Recommendations.Add("Consider server-side ViewState storage");
            }
            else if (result.ViewStateSizeKB > 20)
            {
                result.SecurityRisk = "MEDIUM - Large ViewState (>20KB)";
                result.Recommendations.Add("Review control ViewState usage");
            }
            else
            {
                result.SecurityRisk = "LOW - Acceptable ViewState size";
            }
            
            // Check for potential tampering indicators
            if (HasSuspiciousPatterns(viewStateData))
            {
                result.SecurityRisk = "CRITICAL - Suspicious patterns detected";
                result.Recommendations.Add("IMMEDIATE INVESTIGATION REQUIRED");
            }
            
        }
        catch (Exception ex)
        {
            result.SecurityRisk = "ERROR - ViewState parsing failed";
            result.ErrorMessage = ex.Message;
        }
        
        return result;
    }
    
    private bool HasSuspiciousPatterns(string viewState)
    {
        // Check for known attack patterns
        string[] suspiciousPatterns = 
        {
            "System.Web.UI.ObjectStateFormatter",
            "System.Activities.Presentation.WorkflowDesigner",
            "Microsoft.Exchange.Management",
            "System.Security.Claims.ClaimsIdentity"
        };
        
        foreach (string pattern in suspiciousPatterns)
        {
            if (viewState.Contains(Convert.ToBase64String(Encoding.UTF8.GetBytes(pattern))))
            {
                return true;
            }
        }
        
        return false;
    }
}

public class ViewStateAnalysisResult
{
    public int ViewStateSizeBytes { get; set; }
    public double ViewStateSizeKB { get; set; }
    public string SecurityRisk { get; set; }
    public bool ContainsControlState { get; set; }
    public bool ContainsEventValidation { get; set; }
    public List<string> Recommendations { get; set; } = new List<string>();
    public string ErrorMessage { get; set; }
}
